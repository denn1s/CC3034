React

  • Librería de javascript para construir interfaces
  • No es un framework
  • Permite construir interfaces complejas a partir de pequeñas partes llamadas componentes
  • Se trabaja en JSX

              class ShoppingList extends React.Component {
                render() {
                  return (
                    <div className="shopping-list">
                    <h1>Shopping List for {this.props.name}</h1>
                      <ul>
                        <li>Instagram</li>
                        <li>WhatsApp</li>
                        <li>Oculus</li>
                      </ul>
                    </div>
                  );
                }
              }

Polymer

  • Es una librería para traer web components a la web actual
  • El objetivo de Polymer es desaparecer
  • Se integra con frameworks
  • ES6 + npm

              <body>
                <likeable-element></likeable-element>
              </body>
            

                import '@polymer/paper-checkbox/paper-checkbox.js';

                // Import the PolymerElement base class and html helper
                import {PolymerElement, html} from '@polymer/polymer';
                
                // Define an element class
                class LikeableElement extends PolymerElement {
                
                  // Define publc API properties
                  static get properties() { return { liked: Boolean }}
                
                  // Define the element's template
                  static get template() {
                    return html`
                      <style>
                        .response { margin-top: 10px; } 
                      </style>
                      <paper-checkbox checked="{{liked}}">I like web components.</paper-checkbox>
                
                      <div hidden$="[[!liked]]" class="response">Web components like you, too.</div>
                    `;
                  }
                }
                
                // Register the element with the browser
                customElements.define('likeable-element', LikeableElement);

GraphQL

  • Query language para APIs
  • Permite reformatear data
  • Permite unificar endpoints
  • Permite tener una interfaz que describe el sitema

              var { graphql, buildSchema } = require('graphql');

              // Construct a schema, using GraphQL schema language
              var schema = buildSchema(`
                type Query {
                  hello: String
                }
              `);
              
              // The root provides a resolver function for each API endpoint
              var root = {
                hello: () => {
                  return 'Hello world!';
                },
              };
              
              // Run the GraphQL query '{ hello }' and print out the response
              graphql(schema, '{ hello }', root).then((response) => {
                console.log(response);
              });

Sass

  • Lenguaje de extensión de css
  • No es un framework, pero hay muchos frameworks que lo usan
  • Agrega a css varios features como variables/mixins
  • Agrega algunos features de OO

              $primary-color: #333;

              }

  • No es un framework y hay muy pocos frameworks que lo proveen
  • Mejora considerablemente la experiencia del usuario
  • SEO

PWA

  • Progressive Web Apps
  • Concepto creado por google, y promocionado por google
  • Service workers
  • Cargan instantaneamente, deben estar disponibles offline

              if ('serviceWorker' in navigator) {
                navigator.serviceWorker.register('/sw-test/sw.js', {scope: '/sw-test/'})
                .then(function(reg) {
                  // registration worked
                  console.log('Registration succeeded. Scope is ' + reg.scope);
                }).catch(function(error) {
                  // registration failed
                  console.log('Registration failed with ' + error);
                });
              }

Electron

  • Permite crear aplicaciones para correr en el desktop
  • La aplicación provicionada, idealmente debe ser una PWA
  • Crea paquetes instalables
  • nodejs + chromium

              const { app, BrowserWindow } = require('electron')

              function createWindow () {
                // Create the browser window.
                win = new BrowserWindow({ width: 800, height: 600 })
              
                // and load the index.html of the app.
                win.loadFile('index.html')
              }
              
              app.on('ready', createWindow)
