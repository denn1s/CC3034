Events


              myObject.on('someEvent', () => {
                console.log('an event occurred!');
              });
              // ...
              myObject.emit('event');

Ejemplo EventEmmiter


              const EventEmitter = require('events');

              class MyEmitter extends EventEmitter {}

              const myEmitter = new MyEmitter();
              myEmitter.on('event', () => {
                console.log('an event occurred!');
              });
              myEmitter.emit('event');
